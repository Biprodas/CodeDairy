#include<bits/stdc++.h>
using namespace std;
#define MX 102

int n, m, u, v, w;
int matrix[MX][MX];

void FloydWarshal(){
    for(int k=1;k<=n;k++){
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                matrix[i][j]= min(matrix[i][j], matrix[i][k] + matrix[k][j]);
            }
        }
    }
}
void print(){
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++)
        if(matrix[i][j]==1e9) cout<<"inf"<<"  ";
        else cout<<matrix[i][j]<<"  ";
        cout<<endl;
    }  
}

int main(){
    cin>>n>>m;
    for(int i=0;i<=n;i++){
       for(int j=0;j<=n;j++){
           i==j ? matrix[i][j]=0 : matrix[i][j]= 1e9;
       }
    }
    while(m--){
        cin>>u>>v>>w;
        matrix[u][v]= w;
    }
    print(); cout<<endl;
    FloydWarshal(); cout<<endl;
    print();
    return 0;
}
